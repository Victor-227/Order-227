# 1. 
is_controlled_by_ROOT_or_subject = {
	custom_trigger_tooltip = {
		tooltip = is_controlled_by_ROOT_or_subject
		any_country = {
			controls_state = PREV
			OR = {
				tag = ROOT
				is_subject_of = ROOT
			}
		}
	}
}

# 7. (Advanced Scoping: Asks if the state is a core of the country checking it)
is_core_or_compliance_60 = {
	custom_trigger_tooltip = {
		tooltip = is_core_or_compliance_60_tt
		OR = {
			any_country = {
				controls_state = PREV
				PREV = { is_core_of = PREV }
			}
			compliance > 60
		}
	}
}

# 8. (has_full_control_of_state natively checks control, so it's extra safe)
state_is_fully_controlled_by_ROOT_or_subject = {
	custom_trigger_tooltip = {
		tooltip = state_is_fully_controlled_by_ROOT_or_subject_tt
		OR = {
			is_fully_controlled_by = ROOT
			any_country = {
				is_subject_of = ROOT
				has_full_control_of_state = PREV
			}
		}
	}
}
